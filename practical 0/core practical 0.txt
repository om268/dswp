# Python Introduction Example

# Taking user input
name = input("Enter your name: ")
age = int(input("Enter your age: "))
city = input("Enter your city: ")

# Processing the input
next_year_age = age + 1

# Output the results
print("\n--- User Details ---")
print("Name:", name)
print("Age:", age)
print("City:", city)
print("Next year, you will be", next_year_age, "years old.")
